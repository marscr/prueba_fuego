Problema 1:

Escriba tres funciones que sumen todos los números de la siguiente lista:

Lista:

  • Agregar
  • Eliminar
  • Solución

Solución con ciclo for

    function sumafor(lista){
      var resultado=0;
        for( var i=0 ; i < lista.length ; i++ ){
          resultado += lista[i];
        }//fin del ciclo for que recorre la lista
        return resultado;
    }//fin sumafor

Solución con ciclo while

    function sumawhile(lista){
      var resultado=0,i=0;
        while(i < lista.length){
          resultado+=lista[i];
          i++;
        }
        return resultado;
    }fin sumawhile

Solución con recursividad 1

    //////////////Suma Recursiva que deja intacta la lista
    function sumarecursiva1(lista,i){
      variable = lista[i];
      if(i===lista.length-1){
          //Condición de salida
          return variable;
      }
      else{
        //Llamada recursiva
        return variable+(sumarecursiva1(lista,i+1));
      }
    }//fin sumarecursiva

Solución con recursividad 2

    function sumarecursiva2(lista){
      if(lista.length===0){
        //Lista Vacía
        return 0;
      }
      else if(lista.length===1){
        //Condición de Salida
        return lista[0];
      }
      else{
        //Llamada Recursiva
        lista[0]+=lista.pop();
        return sumarecursiva2(lista);
      }
    }//fin sumarecursiva
